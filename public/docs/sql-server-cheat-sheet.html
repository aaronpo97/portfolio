Basic CRUD (Create, Read, Update, Delete)

Basic INSERT Commands - CREATE

-- Insert a single row
INSERT INTO table_name (column1, column2)
VALUES (value1, value2);

-- Insert multiple rows
INSERT INTO table_name (column1, column2)
VALUES (value1a, value2a),
       (value1b, value2b);

-- Insert from another table
INSERT INTO table_name (column1, column2)
SELECT column1, column2 FROM other_table WHERE condition;

Basic SELECT Commands - READ

SELECT * FROM table_name;
SELECT column1, column2 AS 'Alias' FROM table_name;
SELECT column1 + column2 AS 'Sum' FROM table_name;
SELECT DISTINCT column FROM table_name;
SELECT * FROM table_name ORDER BY column1 ASC, column2 DESC;

String Literals & Concatenation

SELECT FirstName + ' ' + LastName AS FullName FROM Customers;

Numeric Literals & Math Expressions 

SELECT RetailPrice * 1.13 AS PriceWithTax FROM Products;

Basic UPDATE Commands - UPDATE

-- Update a single row
UPDATE table_name
SET column1 = value1, column2 = value2
WHERE condition;

-- Update multiple rows
UPDATE table_name
SET column1 = value1
WHERE column2 > value2;

-- Update using values from another table
UPDATE t1
SET t1.column1 = t2.column2
FROM table1 t1
INNER JOIN table2 t2 ON t1.id = t2.id
WHERE t2.condition = value;

Basic DELETE Commands - DELETE

-- Delete all rows from a table
DELETE FROM table_name;

-- Delete rows matching a condition
DELETE FROM table_name
WHERE condition;

-- Delete using a join (SQL Server syntax)
DELETE t1
FROM table1 t1
INNER JOIN table2 t2 ON t1.id = t2.id
WHERE t2.condition = value;

Filtering Data (WHERE Clause)

Comparison Operators

-- column is
WHERE column = value

-- column not equals
WHERE column <> value
WHERE column != value -- equivalent

-- column is greater than value
WHERE column > value

-- column is less than value
WHERE column < value

Range Operators

-- EQUIVALENT
WHERE column BETWEEN val1 AND val2
WHERE column >= val1 AND column <= val2

-- EQUIVALENT
WHERE column IN ('val1', 'val2', 'val3')
WHERE column = 'val1' OR column = 'val2' OR column = 'val3'

Pattern Matching (LIKE)

WHERE column LIKE 'P%'   -- starts with 'P'
WHERE column LIKE '%e'   -- ends with 'e'
WHERE column LIKE '%e%'  -- contains 'e'
WHERE column LIKE '_e'   -- second char is 'e'
WHERE column LIKE 'P_r%' -- 'P', any char, 'r', then anything

Null Handling

WHERE column IS NULL
WHERE column IS NOT NULL

Logical Operators

WHERE condition1 AND condition2
WHERE condition1 OR condition2
WHERE NOT condition

Order of Precedence - you can also use parentheses to control the order of evaluation

NOT > AND > OR

Filtering Dates

Default format: ‘YYYY-MM-DD’

WHERE OrderDate = '2022-09-02' -- Order date is exactly September 2, 2022
WHERE OrderDate BETWEEN '2022-09-01' AND '2022-09-03' -- Order date is between these two dates
WHERE ShipDate IN ('2022-09-01', '2022-09-02', '2022-09-03') -- Ship date is one of these specific dates
WHERE ShipDate LIKE '2022-09%' -- Ship date is in September 2022
-- use the other comparison operators for other date conditions

JOINS

Cartesian Join

A Cartesian join (or cross join) returns the Cartesian product of two tables, meaning every row from the first table is combined with every row from the second table. 

SELECT * FROM table1 CROSS JOIN table2;

Equality Join (Old Way)

SELECT * FROM table1, table2 WHERE table1.id = table2.id;

INNER JOIN (Preferred Way)

SELECT * FROM table1 INNER JOIN table2 ON table1.id = table2.id;

Multiple Joins

-- Joining three tables A, B, and C
SELECT * FROM A
INNER JOIN B ON A.id = B.id
INNER JOIN C ON B.id = C.id;

Self Join

A self join is a join where a table is joined with itself. This is useful for comparing rows within the same table. 

-- Example: Finding customers with the same last name

SELECT c1.FirstName, c1.LastName, c2.FirstName, c2.LastName
FROM Customers c1
INNER JOIN Customers c2 ON c1.LastName = c2.LastName AND c1.Id <> c2.Id;

DDL: Table Creation

Basic Table

CREATE TABLE test_customer (
    Id INT,
    Name VARCHAR(20)
);

Table with Primary Key (Column Constraint) - Do not do this 

CREATE TABLE test_customer (
    CustID SMALLINT PRIMARY KEY,
    CustName VARCHAR(20)
);

Table with Primary Key (Table Constraint) 

CREATE TABLE test_customer (
    CustID SMALLINT,
    CustName VARCHAR(20),
    CONSTRAINT test_customer_CustID_pk PRIMARY KEY (CustID)
);

Table with Foreign Key

CREATE TABLE test_order (
    OrderID SMALLINT,
    CustID SMALLINT,
    CONSTRAINT test_order_orderID_pk PRIMARY KEY (OrderID),
    CONSTRAINT test_orders_test_customer_fk FOREIGN KEY (CustID) REFERENCES test_customer(CustID)
);

Add Column

Modify Column

ALTER TABLE Vendors ALTER COLUMN Ext INT;

Drop Column

ALTER TABLE Vendors DROP COLUMN Ext;

Constraints

There are 5 types of constraints:

Type Description
PRIMARY KEY Uniquely identifies a row
FOREIGN KEY References key in another table
UNIQUE Ensures unique values in column
CHECK Validates range/logic on column
NOT NULL Column cannot be NULL

CHECK Example

CONSTRAINT test_customer_age_ck CHECK (Age >= 18 AND Age <= 120)

UNIQUE Example

CONSTRAINT test_customer_SIN_UK UNIQUE(SIN)

DROP TABLE

DROP TABLE table_name;

Optional - IF EXISTS

DROP TABLE IF EXISTS table_name;
